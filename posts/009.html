最大匹配问题

大一 图论
算法 最大匹配
发布日期:   2022-04-26
更新日期:   2022-07-11
文章字数:   1.9k
阅读时长:   9 分

引入

前提：讨论的是二分简单图

如何判断是否为二分图

定义：如果某个图为二分图，那么它至少有两个顶点，且其所有回路的长度均为偶数，任何无回路的的图均是二分图。

方法：利用标点（或者染色）法，假设图G有v个结点、e条边，从任一结点出发（其标记为 x ），遍历所有与其邻接的结点，如果其邻接的结点未被访问过，就将其标记为 -x ，如果其邻接的结点被访问过，判断其标号是否为 x ，若为 x 则不是二分图，直至将全部结点都访问，所有邻接的结点的标号是不同的，则该图为二分图。

代码实现：

#include<bits/stdc++.h>
using namespace std;
const int maxn = 207;

int n, m;
vector<int>G[maxn];
int colo[maxn]{};

bool dfs(int u, int c)
{
	colo[u] = c;
	for (int i = 0; i < G[u].size(); i++)
	{
		int v = G[u][i];
		if (colo[v] == c)return 0;
		if (colo[v] == 0 && !dfs(v, (-1 * c)))return 0;
	}
	return 1;
}

signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n >> m;
	for (int i = 1; i <= m; i++)
	{
		int u, v;
		cin >> u >> v;
		G[u].push_back(v);
		G[v].push_back(u);
	}
	bool ok = 1;
	for (int i = 1; i <= n; i++)
	{
		if (!colo[i])
		{
			if (!dfs(i, 1))
			{
				ok = 0;
				break;
			}
		}
	}
	if (ok)cout << "该图是二分图" << endl;
	else cout << "该图不是二分图" << endl;
	return 0;
}

遍历时我们这里选择的是DFS去遍历，当然选择BFS去遍历当然也是没问题的，核心的思路仍旧是染色法。

相关概念

交错路径：给定图G的一个匹配 M ,如果一条路径的边交替出现在M和不出现在M中，则将这条路径称为一条M-交错路径。

增广路径：如果一条交错路径的两个端点是非饱和点（不与匹配的边相关联），我们称这条路径为增广路径。

Hall定理：对于一个二部图G~(X,Y)，X存在一个匹配的充分必要条件为对于X的任意子集S，S的邻居个数N(S)必须大于等于S的大小|S|。

最大匹配：若当前M中不存在增广路径，则M为最大匹配。（对于一个图而言，最大匹配可能不唯一，但最大匹配数是唯一确定的）

匈牙利算法

匈牙利算法的核心思路：在当前已经确定的匹配中，不断寻找增广路径，直至不存在增广路径。

代码实现：“先到先得，能让则让”

#include<bits/stdc++.h>
#define mst(a) memset(a,0,sizeof(a))
using namespace std;
const int maxn = 107;

int n, m; //二分图两个顶点集的大小
bool vis[maxn] {}; //标记B中结点是否被访问
int p[maxn] {}; //当前B中的结点的邻接结点为p[i]
int mp[maxn][maxn] {}; //邻接表存二分图G（A->B）

bool match(int u) {
	for (int v = 1; v <= m; v++) {
		if (mp[u][v] && !vis[v]) {
			vis[v] = 1;
			if (p[v] == 0 || match(p[v])) {
				p[v] = u;
				return 1;
			}
		}
	}
	return 0;
}

void sovle() {
	int cnt = 0;
	for (int i = 1; i <= n; i++) {
		mst(vis); //每次都需要重置vis
		if (match(i))cnt++;
	}
	cout << cnt << endl;
}

仔细体会：“先到先得，能让则让”，在match（）函数的递归过程中，这八个字可以说被体现的淋漓尽致。

HK算法

HK算法（Hopcroft-Carp算法）是对匈牙利算法的一个优化，匈牙利算法时间复杂度为 O（N M）级别的，在1000个点以内时间复杂度尚可接受，但是如果数据变大，时间复杂度无法接受，HK算法就可以将时间复杂度降低至 O（N^（1/2）M）级别。（原因：匈牙利算法一次只更新一条增广路，而HK算法同时更新多条互不相交的最短增广路。时间复杂度不予证明，感兴趣的可以自行去看）

HK算法本质是对匈牙利算法进行了优化，匈牙利是依次去遍历点和边，而HK算法是用BFS同时去寻找多条增广路，然后寻找到了极大增广路集，然后再用匈牙利算法的DFS去对增广路集进行增广，直接上代码吧，可以当板子用。

一般而言，500个点以内选择匈牙利算法，5000个点以内选择HK算法。

例题

HDU2389

AC代码:

#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <queue>
#include <cmath>
using namespace std;
 
const int MAXN = 3010;
const int MAXM = 3010*3010;
const int INF = 0x3f3f3f3f;
 
struct Edge
{
	int v;
	int next;
}edge[MAXM];
 
struct node
{
	double x, y;
	double v;
}a[MAXN], b[MAXN];
 
int nx, ny;
int cnt;
int t;
int dis;
 
int first[MAXN];
int xlink[MAXN], ylink[MAXN]; 
/*xlink[i]表示左集合顶点所匹配的右集合顶点序号，ylink[i]表示右集合i顶点匹配到的左集合顶点序号。*/
int dx[MAXN], dy[MAXN]; 
/*dx[i]表示左集合i顶点的距离编号，dy[i]表示右集合i顶点的距离编号*/
int vis[MAXN]; //寻找增广路的标记数组 
 
void init()
{
	cnt = 0;
	memset(first, -1, sizeof(first));
	memset(xlink, -1, sizeof(xlink));
	memset(ylink, -1, sizeof(ylink));
}
 
void read_graph(int u, int v)
{
	edge[cnt].v = v;
	edge[cnt].next = first[u], first[u] = cnt++;
}
 
int bfs()
{
	queue<int> q;
	dis = INF;
	memset(dx, -1, sizeof(dx));
	memset(dy, -1, sizeof(dy));
	for(int i = 0; i < nx; i++)
	{
		if(xlink[i] == -1)
		{
			q.push(i);
			dx[i] = 0;
		}
	}
	while(!q.empty())
	{
		int u = q.front(); q.pop();
		if(dx[u] > dis) break;
		for(int e = first[u]; e != -1; e = edge[e].next)
		{
			int v = edge[e].v;
			if(dy[v] == -1)
			{
				dy[v] = dx[u] + 1;
				if(ylink[v] == -1) dis = dy[v];
				else
				{
					dx[ylink[v]] = dy[v]+1;
					q.push(ylink[v]);
				}
			}
		}
	}
	return dis != INF;
}
 
int find(int u)
{
	for(int e = first[u]; e != -1; e = edge[e].next)
	{
		int v = edge[e].v;
		if(!vis[v] && dy[v] == dx[u]+1)
		{
			vis[v] = 1;
			if(ylink[v] != -1 && dy[v] == dis) continue;
			if(ylink[v] == -1 || find(ylink[v]))
			{
				xlink[u] = v, ylink[v] = u;
				return 1;
			}
		}
	}
	return 0;
}
 
int MaxMatch()
{
	int ans = 0;
	while(bfs())
	{
		memset(vis, 0, sizeof(vis));
		for(int i = 0; i < nx; i++) if(xlink[i] == -1)
		{
			ans += find(i);
		}
	}
	return ans;
}
 
double dist(const node a, const node b)
{
	return sqrt((a.x-b.x)*(a.x-b.x) + (a.y-b.y)*(a.y-b.y));
}
 
void read_case()
{
	init();
	int Time;
	scanf("%d", &Time);
	scanf("%d", &nx);
	for(int i = 0; i < nx; i++)
	{
		scanf("%lf%lf%lf", &a[i].x, &a[i].y, &a[i].v);
	}
	scanf("%d", &ny);
	for(int i = 0; i < ny; i++)
	{
		scanf("%lf%lf", &b[i].x, &b[i].y);
	}
	for(int i = 0; i < nx; i++)
	{
		for(int j = 0; j < ny; j++)
		{
			double limit = a[i].v*Time;
			double s = dist(a[i], b[j]);
			if(s <= limit) read_graph(i, j);
		}
	}
}
 
void solve()
{
	read_case();
	int ans = MaxMatch();
	printf("%d\n\n", ans);
}
 
int main()
{
	int T, times = 0;
	scanf("%d", &T);
	while(T--)
	{
		printf("Scenario #%d:\n", ++times);
		solve();
	}
	return 0;
}
