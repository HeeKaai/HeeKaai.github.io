01背包

背景描述：

有 N 件物品和一个容量是 V 的背包。每件物品只能使用一次。

第 $i$ 件物品的体积是 $v_i$ ，价值是 $w_i$

求解将哪些物品装入背包，可使这些物品的总体积不超过背包容量，且总价值最大。

代码：

#define mst(a) memset(a,0,sizeof(a))
const int maxn = 1007;

void solve()
{
	int N, V, w[maxn], v[maxn];
	cin >> N >> V;
	for (int i = 1; i <= N; i++)
		cin >> v[i] >> w[i];
	int dp[maxn];
	mst(dp);
	for (int i = 1; i <= N; i++)
		for (int j = V; j >= v[i]; j--)
			dp[j] = max(dp[j], dp[j - v[i]] + w[i]);
	cout << dp[V] << endl;
}

完全背包

背景描述：

有 $N$ 种物品和一个容量是 $V$ 的背包，每种物品都有无限件可用。

第 $i$ 种物品的体积是 $v_i$，价值是 $w_i$。

求解将哪些物品装入背包，可使这些物品的总体积不超过背包容量，且总价值最大。

代码：

#define mst(a) memset(a,0,sizeof(a))
const int maxn = 1007;

int N, V;
int dp[maxn], w[maxn], v[maxn];

void solve()
{
	mst(dp);
	cin >> N >> V;
	for (int i = 1; i <= N; i++)
		cin >> v[i] >> w[i];
	for (int i = 1; i <= N; i++)
		for (int j = v[i]; j <= V; j++)
			dp[j] = max(dp[j], dp[j - v[i]] + w[i]);
	cout << dp[V] << endl;
}

模板题链接：完全背包问题

变式链接：AcWing279自然数拆分（最大价值->方案数）

变式：AcWing280陪审团

解题思路：

1、状态设计：$dp[i][j][k]$ — 前 $i$ 个人中选择了 $j$ 个人作为陪审团 并且D与P的差值为$k$ 的方案（也即满足差值为k且D与P的和最大）

2、状态转移：

3、状态回溯找转移的路径

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 2147483648;
const int maxn = 207;
const double eps = 1e-6;

int n, m;
int p[maxn], d[maxn];
int dp[maxn][27][807];
int base = 400;
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	int T = 1;
	while (cin >> n >> m)
	{
		if (!n && !m)break;
		cout << "Jury #" << T << endl;
		T++;
		for (int i = 1; i <= n; i++)
			cin >> p[i] >> d[i];
		mstmin(dp);
		dp[0][0][base] = 0;
		for (int i = 1; i <= n; i++)
		{
			for (int j = 0; j <= m; j++)
			{
				for (int k = 0; k < 807; k++)
				{
					dp[i][j][k] = dp[i - 1][j][k];
					int gap = p[i] - d[i];
					if (j - 1 < 0)continue;
					if (k - gap < 0 || k - gap>800)continue;
					dp[i][j][k] = max(dp[i][j][k], dp[i - 1][j - 1][k - gap] + p[i] + d[i]);
				}
			}
		}
		int v = 0;
		while (dp[n][m][base + v] < 0 && dp[n][m][base - v] < 0)v++;

		if (dp[n][m][base + v] >= dp[n][m][base - v])
			v = v + base;
		else v = base - v;


		vector<int>ans;
		int i = n, j = m;
		while (j > 0)
		{
			if (dp[i][j][v] == dp[i - 1][j][v])
				i--;
			else {
				ans.push_back(i);
				v -= (p[i] - d[i]);
				i--; j--;
			}
		}

		int sp = 0, sd = 0;
		for (int k = 0; k < ans.size(); k++)
		{
			sp += p[ans[k]];
			sd += d[ans[k]];
		}
		cout << "Best jury has value " << sp
			<< " for prosecution and value " << sd
			<< " for defence:" << endl;
		sort(ans.begin(), ans.end());
		for (int k = 0; k < ans.size(); k++)
			cout << " " << ans[k];
		cout << endl << endl;
	}
	return 0;
}

多重背包Ⅰ

背景描述：

有 $N$ 种物品和一个容量是 $V$ 的背包。

第 $i$ 种物品最多有 $si$ 件，每件体积是 $vi$，价值是 $wi$。

求解将哪些物品装入背包，可使物品体积总和不超过背包容量，且价值总和最大。

代码：（二进制优化版）

本质：利用二进制拆分成很多个01背包

#define mst(a) memset(a,0,sizeof(a))
const int maxn = 2007;

int N, V;
int dp[maxn], s[maxn], v[maxn], w[maxn];

void solve()
{
	mst(dp);
	cin >> N >> V;
	for (int i = 1; i <= N; i++)
		cin >> v[i] >> w[i] >> s[i];
	for (int i = 1; i <= N; i++)
	{
		int res = s[i];
		for (int k = 1; res >= k;)
		{
			for (int j = V; j >= v[i] * k; j--)
				dp[j] = max(dp[j], dp[j - v[i] * k] + k * w[i]);
			res -= k;
			k *= 2;
		}
		for (int j = V; j >= v[i] * res; j--)
			dp[j] = max(dp[j], dp[j - v[i] * res] + res * w[i]);
	}
	cout << dp[V] << endl;
}

多重背包Ⅱ

背景描述：

描述和多重背包Ⅰ一致，只是把数据范围扩大

引入：单调队列的维护

模板题链接：单调队列

很简单，就是每次输出队首元素，维护的时候队尾删去比当前小的，队首删去时间不符合要求的即可。

AC代码：

void solve()
{
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i] >> b[i];
	int l = 1, r = 0;
	for (int i = 1; i <= n; i++)
	{
		for (; l <= r && b[i] >= b[q[r]]; r--);
		r++; q[r] = i;
		cout << b[q[l]] << endl;
		for (; l <= r && i == a[q[l]]; l++);
	}
}

代码（单调队列优化）:

int n, m, v, w, t, f[10001], c[10001][2];

void solve()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		scanf("%d%d%d", &v, &w, &t);
		for (int j = 0; j < v; j++) {
			int k = 0, l = 1;
			for (int p = j, x = 1; p <= m; p += v, ++x) {
				int e = f[p] - x * w, r = x + t;
				for (; k >= l && c[k][0] <= e; --k);
				c[++k][0] = e; c[k][1] = r;
				f[p] = c[l][0] + x * w;
				for (; k >= l && c[l][1] == x; ++l);
			}
		}
	}
	printf("%d\n", f[m]);
}

分组背包

背景描述：

有 $N$ 组物品和一个容量是 $V$ 的背包。

每组物品有若干个，同一组内的物品最多只能选一个。
每件物品的体积是 $v[i][j]$，价值是 $w[i][j]$，其中 $i$ 是组号，$j$ 是组内编号。

求解将哪些物品装入背包，可使物品总体积不超过背包容量，且总价值最大。

代码：

#define mst0(a) memset(a,0,sizeof(a))
const int maxn = 107;

int N, V;
int s[maxn], dp[maxn];
vector<int>v[maxn], w[maxn];

void solve()
{
	cin >> N >> V;
	mst0(dp);
	for (int i = 1; i <= N; i++)
	{
		cin >> s[i];
		int x, y;
		for (int j = 1; j <= s[i]; j++)
		{
			cin >> x >> y;
			v[i].push_back(x);
			w[i].push_back(y);
		}
	}
	for (int i = 1; i <= N; i++)
	{
		for (int j = V; j >= 0; j--)
		{
			for (int k = 0; k < s[i]; k++)
			{
				if (j >= v[i][k])
					dp[j] = max(dp[j], dp[j - v[i][k]] + w[i][k]);
			}
		}
	}
	cout << dp[V] << endl;
}

二维（费用）背包

背景描述：

有 $N$ 件物品和一个容量是 $V$ 的背包，背包能承受的最大重量是 $M$。

每件物品只能用一次。体积是 $v_i$，重量是 $m_i$ ，价值是 $w_i$ 。

求解将哪些物品装入背包，可使物品总体积不超过背包容量，总重量不超过背包可承受的最大重量，并且要使得得到的价值总和最大。

输出最大价值。

01背包扩展一个维度即可，很好理解。

代码：

#define mst0(a) memset(a,0,sizeof(a))
const int maxn = 1007;

int N, V, M;
int v[maxn], m[maxn], w[maxn];
int dp[107][107];

void solve()
{
	cin >> N >> V >> M;
	for (int i = 1; i <= N; i++)
		cin >> v[i] >> m[i] >> w[i];
	for (int i = 1; i <= N; i++)
	{
		for (int j = V; j >= v[i]; j--)
		{
			for (int k = M; k >= m[i]; k--)
			{
				dp[j][k] = max(dp[j][k], dp[j - v[i]][k - m[i]] + w[i]);
			}
		}
	}
	cout << dp[V][M] << endl;
}

CF366C

题目大意：冰箱里有 n 个水果, 每个水果都有两个参数: 味道和热量。现在从冰箱里拿出一些水果，来做水果沙拉。拿的水果必须遵循:所选的水果总味道与总热量之比必须等于 k。给了每个水果的味道 $aj$ ，热量 $bj$ 。让我们找到满足上述原则时，选出的水果总味道之和的最大值。

解题思路：

1、明显是 01背包 的变形

2、先处理集合 b：$b[i]=>a[i]-b[i]*k$ ，这样处理之后只要保证从集合 b 中选取的数之和为 0 即可

3、明显处理后的 $b[i]$ 有正有负，分别求得前 n 件水果的 b（绝对值）之和能达到的最大的 a之和。

4、最后遍历，把b（绝对值）之和相同的得到的求最大值即可

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 2147483648;
const int maxn = 107;
const double eps = 1e-6;

int n, k;
int a[maxn], b[maxn], c[maxn];
int dp[2][10007];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n >> k;
	int V = 0;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i];
		V += a[i];
	}
	for (int i = 1; i <= n; i++)
	{
		cin >> b[i];
		b[i] = a[i] - b[i] * k;
	}
	mstmin(dp);
	dp[0][0] = dp[1][0] = 0;
	for (int i = 1; i <= n; i++)
	{
		if (b[i] >= 0)
		{
			for (int j = V; j >= b[i]; j--)
				dp[0][j] = max(dp[0][j], dp[0][j - b[i]] + a[i]);
		}
		else {
			int xx = b[i] * -1;
			for (int j = V; j >= xx; j--)
				dp[1][j] = max(dp[1][j], dp[1][j + b[i]] + a[i]);
		}
	}
	int ans = 0;
	for (int j = 0; j <= V; j++)
		ans = max(ans, dp[0][j] + dp[1][j]);
	if (!ans)ans = -1;
	cout << ans << endl;
	return 0;
}

CF864E

题目大意： 一个房子起火了，房子里面有 $n$ 个物品，第 $i$ 件物品价值 $pi$ 块，需要花费 $ti$ 的时间才能救出这个物品，并且这个物品会在 $di$ 时刻之后被烧毁，也就是过了$di$ 时间之后再来救这个物品，它就没有任何的价值了。输出救出物品最大价值总和，以及救出物品的顺序。

解题思路：

1、很明显的 01背包 + 路径问题

2、路径问题要涉及所有的状态，所以状态设计不能压缩

3、这题关键在于对 d 的排序，因为转移的时候大的是由小的转移过来的，则需要先把所有小于 $d[i]$ 的状态求解出来，则用一个结构体存储物品，需要记录原来的位置（路径输出的时候要输出原来的位置）

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 2147483648;
const int maxn = 107;
const double eps = 1e-6;

int n;
struct good
{
	int d, p, t;
	int id;
}G[maxn];
int dp[maxn][2007];
bool cmp(good g1, good g2) {
	return g1.d < g2.d;
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> G[i].t >> G[i].d >> G[i].p;
		G[i].id = i;
	}
	sort(G + 1, G + 1 + n, cmp);
	mst0(dp);
	int ans = -1;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 0; j <= G[i].d; j++)
		{
			dp[i][j] = dp[i - 1][j];
			if (j > G[i].t)
			{
				dp[i][j] = max(dp[i][j], dp[i - 1][j - G[i].t] + G[i].p);
			}
			ans = max(ans, dp[i][j]);
		}
	}
	int i = n, j = 0;
	for (; j < 2007; j++)
	{
		if (dp[i][j] == ans)break;
	}
	vector<int>vec;
	while (i > 0)
	{
		if (dp[i][j] == dp[i - 1][j])i--;
		else {
			vec.push_back(i);
			j -= G[i].t;
			i--;
		}
	}
	sort(vec.begin(), vec.end());
	cout << ans << endl;
	cout << vec.size() << endl;
	for (int i = 0; i < vec.size(); i++)
	{
		if (i == vec.size() - 1)
			cout << G[vec[i]].id << endl;
		else cout << G[vec[i]].id << " ";
	}
	return 0;
}

CF543A

题目大意：$n$ 个程序员完成一个项目，这个项目有 $m$ 行的代码。第 $i$ 个程序员写一行代码会产生 $a_i$ 个bug。按照计划，第 $i$ 个程序员写 $v_i$ 行的代码，总共要写 $m$ 行。如果产生的代码bug数 <= b, 则可以称之为好计划，问是好计划的方案数。

解题思路：

1、完全背包求解方案数

2、状态设计：$dp(i)[j][k]$ — 前 $i$ 个人写了 $j$ 行代码，产生 $k$ 个 bug 的方案数。因为转移时可把第一维度省略（完全背包的特性）

3、状态转移方程：$dp(i)[j][k]+=dp(i-1)[j-1][k-a[i]]$

4、注意取余和转移条件的细节即可

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 2147483648;
const int maxn = 507;
const double eps = 1e-6;

int n, m, b, t;
int a[maxn];
int dp[maxn][maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n >> m >> b >> t;
	mst0(dp);
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	dp[0][0] = 1;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			for (int k = a[i]; k <= b; k++)
			{
				dp[j][k] += dp[j - 1][k - a[i]];
				dp[j][k] %= t;
			}
		}
	}
	int ans = 0;
	for (int k = 0; k <= b; k++)
	{
		ans += dp[m][k];
		ans %= t;
	}
	cout << ans << endl;
	return 0;
}
