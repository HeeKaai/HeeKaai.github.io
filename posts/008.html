引入

在讲矩阵快速幂的时候，不可避免的需要提及的就是矩阵的运算和快速幂的运用，矩阵快速幂就是把这两者的思想相结合罢了。

矩阵的运算

当然数学原理需要用到线性代数的知识，在此不做赘述，不熟悉矩阵乘法的数学原理的话可以自行去百度。

代码：

// 矩阵乘法（默认为方阵）
#define mst(a) memset(a,0,sizeof(a))
const int maxn = 107;

void matrix_mul(int a[][maxn], int b[][maxn], int n)
{
	mst(temp);
	for (int i = 1; i <= n; i++)
	for (int j = 1; j <= n; j++)
	for (int k = 1; k <= n; k++){
		temp[i][j] = temp[i][j] + (a[i][k] * b[k][j]);
	}
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			a[i][j] = temp[i][j];
	return;
}

快速幂

在这篇博客中提到过快速幂的模板：点击跳转

这里直接放模板，就不再讲原理了：

typedef long long ll;
const int mod = 1000000007;

ll ksm(ll base, ll power, ll mod)
{
	ll res = 1;
	while (power)
	{
		if (power % 2)
		{
			res *= base;
			res %= mod;
		}
		power /= 2;
		base *= base;
		base %= mod;
	}
	return res;
}

矩阵快速幂

有了引入部分的基础，矩阵快速幂的实现就非常简单了。和快速幂区别只是把底数换成了矩阵罢了。

代码实现：

#define mst(a) memset(a,0,sizeof(a))
const int mod = 1000000007;
const int maxn = 1007;

ll temp[maxn][maxn]{};
ll res[maxn][maxn]{};

void matrix_mul(ll a[][maxn], ll b[][maxn], ll n)
{
	mst(temp);
	for (int i = 1; i <= n; i++)
	for (int j = 1; j <= n; j++)
	for (int k = 1; k <= n; k++){
		temp[i][j] = temp[i][j] + (a[i][k] * b[k][j]);
		temp[i][j] %= mod;
	}
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			a[i][j] = temp[i][j];
	return;
}

void matrix_ksm(ll a[][maxn], ll power, ll n)
{
	mst(res);
	for (int i = 1; i <= n; i++)
		res[i][i] = 1;
	while (power)
	{
		if (power & 1)
		{
			matrix_mul(res, a, n);
		}
		power /= 2;
		matrix_mul(a, a, n);
	}
}

模板题：矩阵快速幂模板

应用

1、原题链接

题目给出了递推式：$f_i=f_{i-1}+f_{i+1}$ ,对递推式稍加变形也即：$f_i=f_{i-1}-f_{i-2}$

（本题可以由递推式得到循环节，从而通过循环节来解决该问题，但我们关注的是其利用矩阵快速幂进行解题）

推导：

有了如上的推导式，再利用上面提到的矩阵快速幂的思想，就能够轻松的AC此题了。

代码实现如下：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
typedef long long ll;
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 107;
const double eps = 1e-11;

ll temp[maxn][maxn]{};
ll res[maxn][maxn]{};

void matrix_mul(ll a[][maxn], ll b[][maxn], ll n)
{
	mst(temp);
	for (int i = 1; i <= n; i++)
	for (int j = 1; j <= n; j++)
	for (int k = 1; k <= n; k++){
		temp[i][j] = temp[i][j] + (a[i][k] * b[k][j]);
		temp[i][j] %= mod;
	}
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			a[i][j] = temp[i][j];
	return;
}

void matrix_ksm(ll a[][maxn], ll power, ll n)
{
	mst(res);
	for (int i = 1; i <= n; i++)
		res[i][i] = 1;
	while (power)
	{
		if (power & 1)
		{
			matrix_mul(res, a, n);
		}
		power /= 2;
		matrix_mul(a, a, n);
	}
}

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll x, y, k;
	cin >> x >> y >> k;
	if (k <= 2)
	{
		if (k == 1)
			cout << (x + mod) % mod << endl;
		else cout << (y + mod) % mod << endl;
		return 0;
	}
	ll a[maxn][maxn]{};
	a[1][1] = a[1][2] = 1;
	a[2][1] = -1; a[2][2] = 0;
	matrix_ksm(a, (k - 2), 2);
	ll ans = y * res[1][1] + x * res[2][1];
	cout << (ans + mod*2) % mod << endl;
	return 0;
}

2、原题链接

题面描述：原题描述的非常清楚没有什么地方需要解释的

递推：

利用递推公式和快速幂的知识，就能够轻松的解决此题了

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
typedef long long ll;
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 107;
const double eps = 1e-11;

int k, m;
int xx[maxn][maxn]{}, f[13]{};
int res[maxn][maxn]{}, temp[maxn][maxn]{};

void matrix_mul(int a[][maxn], int b[][maxn], int n)
{
	mst(temp);
	for(int i=1;i<=n;i++)
	for(int j=1;j<=n;j++)
		for (int k = 1; k <= n; k++)
		{
			temp[i][j] += (a[i][k] * b[k][j]);
			temp[i][j] %= m;
	}
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			a[i][j] = temp[i][j];
	return;
}

void matrix_ksm(int a[][maxn], int power, int n)
{
	mst(res);
	for (int i = 1; i <= n; i++)
		res[i][i] = 1;
	while (power)
	{
		if (power & 1)
			matrix_mul(res, a, n);
		power /= 2;
		matrix_mul(a, a, n);
	}
}

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	for (int i = 1; i <= 10; i++)
		f[i] = 10 - i;
	while (cin >> k >> m)
	{
		mst(xx);
		for (int i = 0; i <= 9; i++)
			cin >> xx[i + 1][1];
		for (int i = 1; i <= 9; i++)
			xx[i][i + 1] = 1;
		if (k < 9)
		{
			cout << (k + m) % m << endl;
			continue;
		}
		matrix_ksm(xx, k - 9, 10);
		int ans = 0;
		for (int i = 1; i <= 10; i++)
		{
			ans += (res[i][1] * f[i]);
			ans %= m;
		}
		cout << ans << endl;
	}
	return 0;
}

3、原题链接HDU-1588

思路：题目意思非常简洁明了，这题用到矩阵快速幂的知识很简单（表示斐波那契数列的第n项，有上面的基础很容易表示，在这里不做出推导了），第二个考点无非就是等比数列求和的矩阵形式，同样可以用二分解决，下面的图片是核心思路。

（斐波那契数列矩阵表示如下）

（用二分解决等比数列求和，在这里每一项改成矩阵罢了）

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 7;
const double eps = 1e-11;

int k, b, n, m;
struct Mat {
	int a[maxn][maxn]{};
};

Mat mul(Mat m1, Mat m2) {
	Mat ans;
	mst(ans.a);
	for (int i = 1; i <= 2; i++)
		for (int j = 1; j <= 2; j++)
			for (int k = 1; k <= 2; k++) {
				ans.a[i][j] += m1.a[i][k] * m2.a[k][j];
				ans.a[i][j] %= m;
			}
	return ans;
}

Mat ksm(Mat m, int power) {
	Mat ans;
	mst(ans.a);
	for (int i = 1; i <= 2; i++)ans.a[i][i] = 1;
	while (power) {
		if (power & 1)ans = mul(ans, m);
		power >>= 1;
		m = mul(m, m);
	}
	return ans;
}

Mat add(Mat m1, Mat m2) {
	Mat ans;
	mst(ans.a);
	for (int i = 1; i <= 2; i++)
		for (int j = 1; j <= 2; j++) {
			ans.a[i][j] = m1.a[i][j] + m2.a[i][j];
			ans.a[i][j] %= m;
		}
	return ans;
}

Mat sum(Mat m, int n) {
	if (n == 1)return m;
	if (n & 1) {
		return add(sum(m, n - 1), ksm(m, n));
	}
	else {
		Mat tmp;
		tmp = sum(m, n / 2);
		return add(tmp, mul(tmp, ksm(m, n / 2)));
	}
}

signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	while (cin >> k >> b >> n >> m) {
		Mat E;
		mst(E.a);
		E.a[1][1] = E.a[2][2] = 1;
		Mat A;
		mst(A.a);
		A.a[1][1] = A.a[2][1] = A.a[1][2] = 1;
		Mat ans = mul(ksm(A, b), add(E, sum(ksm(A, k), n - 1)));
		cout << ans.a[1][2] << endl;
	}
	return 0;
}
