树形DP

大一 假期 DP
算法 动态规划
发布日期:   2022-07-08
更新日期:   2022-07-24
文章字数:   3.5k
阅读时长:   17 分

简介

树形 DP，也就是在树上进行的 DP。由于树固有的递归性质，树形 DP 一般都是递归进行的。因为给定的树通常是无根树（$N-1$ 条无向边），我们可以任意选择节点作为根节点，从而确定一棵有根树，根据其父亲节点与儿子节点直接的关系设计动态规划对问题进行求解。

故递归和回溯是实现树形DP的基础和关键。

设计状态时，通常将节点的编号设计为第一维度。

入门题

AcWing285没有上司的舞会

解题思路：

1、状态设计：

2、$x$ 参加舞会时其下属都不能参加舞会，而 $x$ 不参加舞会时，其下属可以选择参加舞会也可以选择不参加舞会

3、状态转移方程：

4、转移很好理解，最终的答案就是 $max(dp[root][0],dp[rooot][1])$

5、DFS、BFS对树进行自顶向下的递归遍历（这也是关键和难点）

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 1000000007;
const int maxn = 6007;
const double eps = 1e-6;

int n;
int H[maxn], dp[maxn][2];
bool have_p[maxn];
vector<int>son[maxn];
void dfs(int x)
{
	dp[x][0] = 0;
	dp[x][1] = H[x];
	for (int i = 0; i < son[x].size(); i++)
	{
		int y = son[x][i];
		dfs(y);
		dp[x][0] += max(dp[y][0], dp[y][1]);
		dp[x][1] += dp[y][0];
	}
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> H[i];
	for (int i = 1; i < n; i++)
	{
		int x, y;
		cin >> x >> y;
		have_p[x] = 1;
		son[y].push_back(x);
	}
	int root;
	for (int i = 1; i <= n; i++)
	{
		if (!have_p[i])
		{
			root = i;
			break;
		}
	}
	dfs(root);
	cout << max(dp[root][0], dp[root][1]) << endl;
	return 0;
}

AcWing288选课

解题思路：

1、树上背包（背包与树形DP的结合，通常看作分组背包的转移）

2、状态设计：$dp[i][j]$ — 以 $i$ 为根的子树中选择了 $j$ 门课程所能够获得的最高的学分（注意其中第 $i$ 门课程一定要选）

3、分组背包的模型：根节点的每一子树都看成一组，这一组内最多只能选择一个

4、特殊处理：课程构成的是森林结构，此时构造一个虚拟的根节点，作为所有没有先修课程的先修课程，此时就能够把森林转变成一棵树

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 1000000007;
const int maxn = 307;
const double eps = 1e-6;

int n, m, dp[maxn][maxn], w[maxn];
vector<int>son[maxn];
void dfs(int u)
{
	for (int i = 0; i < son[u].size(); i++)
	{
		int v = son[u][i];
		dfs(v);
		for (int j = m - 1; j >= 0; j--)
		{
			for (int k = j; k >= 0; k--)
			{
				dp[u][j] = max(dp[u][j], dp[u][j - k] + dp[v][k]);
			}
		}
	}
	for (int i = m; i >= 1; i--)
		dp[u][i] = dp[u][i - 1] + w[u];
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
	{
		int x, y;
		cin >> x >> y;
		son[x].push_back(i);
		w[i] = y;
	}
	m++; w[0] = 0;
	dfs(0);
	cout << dp[0][m] << endl;
	return 0;
}

POJ3585Accumulation Degree

解题思路：

1、二次扫描与换根法

2、$dp[u]$ — 表示以 $u$ 为根的子树中，从 $u$ 出发流向子树的最大流量

3、不妨以 1 为根节点建立有根树，通过自底向上的遍历顺序来得到 dp 数组，转移方程如下：

4、解题重难点在于使用换根法代替暴力枚举源点

5、$F[u]$ — 表示以$u$ 为源点，流经整棵树的最大流量。

6、转移：（仔细思考怎么来的）

7、由转移方程，自顶向下遍历整棵树即可

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 1000000007;
const int maxn = 200007;
const double eps = 1e-6;

int t, n;
vector<pii>G[maxn];
int dp[maxn], F[maxn];
void dfs1(int u, int fa)
{
	for (auto p : G[u])
	{
		int v = p.first;
		if (v == fa)continue;
		dfs1(v, u);
		if (G[v].size() == 1)dp[u] += p.second;
		else dp[u] += min(p.second, dp[v]);
	}
}
void dfs2(int u, int fa)
{
	for (auto p : G[u])
	{
		int v = p.first;
		if (v == fa)continue;
		if (G[u].size() == 1)F[v] = dp[v] + p.second;
		else {
			int x = min(F[u] - min(dp[v], p.second), p.second);
			F[v] = dp[v] + x;
		}
		dfs2(v, u);
	}
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> t;
	while (t--)
	{
		cin >> n;
		for (int i = 1; i <= n; i++)
		{
			G[i].clear();
			dp[i] = 0;
			F[i] = 0;
		}
		for (int i = 1; i < n; i++)
		{
			int x, y, z;
			cin >> x >> y >> z;
			G[x].push_back(pii(y, z));
			G[y].push_back(pii(x, z));
		}
		dfs1(1, -1);
		F[1] = dp[1];
		dfs2(1, -1);
		int ans = -1;
		for (int i = 1; i <= n; i++)
			ans = max(ans, F[i]);
		cout << ans << endl;
	}
	return 0;
}

练习题

HDOJ2196

题目大意：一棵树，求解节点 $i$ 到其最远一个节点的距离

解题思路：

1、数据结构：邻接表存储树，$vectorG[maxn]$ ，pair的第一维是邻接的结点，第二维是边的长度

2、状态设计：$pii\ dp[maxn][2]$

3、状态设计清楚的话，转移非常好写，没有难点，跑两遍$dfs$即可

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 1000000007;
const int maxn = 10007;
const double eps = 1e-6;

int n;
vector<pii>G[maxn];
pii dp[maxn][2];
void dfs1(int u, int fa)
{
	for (auto p : G[u])
	{
		int v = p.first, w = p.second;
		if (v == fa)continue;
		dfs1(v, u);
		if (dp[v][0].first + w > dp[u][1].first)
		{
			dp[u][1].first = dp[v][0].first + w;
			dp[u][1].second = v;
			if (dp[u][1].first > dp[u][0].first)
				swap(dp[u][1], dp[u][0]);
		}
	}
}
void dfs2(int u, int fa)
{
	for (auto p : G[u])
	{
		int v = p.first, w = p.second;
		if (v == fa)continue;
		if (dp[u][0].second != v)
		{
			int x = dp[u][0].first + w;
			if (x > dp[v][1].first)
			{
				dp[v][1].first = x;
				dp[v][1].second = u;
				if (dp[v][1].first > dp[v][0].first)
					swap(dp[v][0], dp[v][1]);
			}
		}
		else {
			int x = dp[u][1].first + w;
			if (x > dp[v][1].first)
			{
				dp[v][1].first = x;
				dp[v][1].second = u;
				if (dp[v][1].first > dp[v][0].first)
					swap(dp[v][0], dp[v][1]);
			}
		}
		dfs2(v, u);
	}
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	while (cin >> n)
	{
		for (int i = 1; i <= n; i++)
		{
			G[i].clear();
			dp[i][0] = dp[i][1] = pii(0, 0);
		}
		for (int i =2; i <= n; i++)
		{
			int u, v = i, w;
			cin >> u >> w;
			G[u].push_back(pii(v, w));
			G[v].push_back(pii(u, w));
		}
		dfs1(1, -1);
		dfs2(1, -1);
		for (int i = 1; i <= n; i++)
			cout << dp[i][0].first << endl;
	}
	return 0;
}

CF791D

题目大意：给定一颗 $n$ 个节点的无根树和每次能够跳 $k$ 步，求解 $\sum f(s,t)\ (s<t)$ ，其中 $f(s,t)=\lceil dis(s,t)/k \rceil$

解题思路：

1、令 $L(u,v)$ 为 $u$ 到 $v$ 的距离偏移量，则有 $f(s,t)=dis(s,t)+L(s,t)/k$

2、问题即可转变成为求解： $\sum dis(u,v) 、\sum L(u,v)$ 即可

3、 $\sum dis(u,v)$ 很好求解，计算每条边的贡献，$x$ 为边的孩子节点，$sum[x]$ 为子树大小，贡献则为：$sum[x]*(n-sum[x])$

4、难点在于对 $\sum L(u,v)$ 的求解。定义 $dp[x][j]$ 为以 $i$ 为根的子树中所有路径长度 $\%k$ 为 $j$ 的条数，还是考虑每条边，设每条边的孩子节点为 $v$ 父亲节点为 $u$ ，则路径长膜为 $(i+j+1)\%k$ 的条数为 $dp[u][i]*dp[v][j]$ 条，此时该边贡献的偏移量应该为：$(k-(i+j+1)\%k)\%k$ ，并且dp的转移方程为 $dp[u][(i+1)\%k]+=dp[v][i]$

5、代码不难写，贵在理解偏移量的求解过程是如何实现的

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 1000000007;
const int maxn = 200007;
const double eps = 1e-6;
inline int read() {
	int x = 0, f = 1;
	char c = getchar();
	while (c < '0' || c > '9') {
		if (c == '-')f = -1;
		c = getchar();
	}
	while (c >= '0' && c <= '9') {
		x = x * 10 + c - '0';
		c = getchar();
	}
	return x * f;
}

int n, k, sum[maxn], dp[maxn][6];
vector<int>G[maxn];
int ans = 0;
void dfs(int u, int fa)
{
	sum[u] = 1;
	dp[u][0] = 1;
	for (auto v : G[u])
	{
		if (v == fa)continue;
		dfs(v, u);
		sum[u] += sum[v];
		ans += (sum[v] * (n - sum[v]));
		for (int i = 0; i < k; i++)
			for (int j = 0; j < k; j++)
			{
				int x = (k - (i + j + 1) % k) % k;
				ans += x * dp[u][i] * dp[v][j];
			}
		for (int j = 0; j < k; j++)
			dp[u][(j + 1) % k] += dp[v][j];
	}
}
signed main() {
	n = read(); k = read();
	for (int i = 1; i < n; i++)
	{
		int u = read(), v = read();
		G[u].push_back(v);
		G[v].push_back(u);
	}
	dfs(1, -1);
	printf("%lld\n", (ans / k));
	return 0;
}

CF708C

题目大意：定义树的质心：假设一棵树有 $n$ 个顶点，对质心（顶点）而言，每颗子树的大小不超过 $n/2$ 即称此顶点为质心，现请问对每个顶点可以进行最多一次操作（删一条边加一条边），能否将其变为质心。

解题思路：

1、换根树形DP

2、先跑一遍 $dfs$ 找出一个质心（可以证明每棵树至少存在一个质心）

3、以质心为根，转变为一颗有根树。定义 $sz[x]$ 为以 $x$ 为根的子树（包括顶点 $x$ ）的大小， $f[x][0/1]$ 以 $x$ 为根的子树中（不包括自身）最大的和次大的 $sz[y]$ ，不用考虑是否超过 $n/2$ ，因为是以质心为根，由上而下子树的大小只会逐渐变小，跑第二遍 $dfs$ 即可实现

4、因为是以质心为根，保证了子树的大小一定不会超过 $n/2$ ，如果 $v$ 顶点的一颗子树大小超过了 $n/2$ ，只能说明它的父亲顶点 $u$ 那一侧出了问题，如何修改是最优的呢，当然是选择最大的且不超 $n/2$ 的一颗子树与当前顶点建边，所以由上至下递归的时候，找出最大的一颗子树， $f[x][0/1]$ 的作用就是判断当前子树是否是最大子树，如果是只能求其次找找第二大的子树，当然每次 $n-sz[v]$ 也要记得作为子树之一作为更新。

5、详情实现见代码，第四点的需要好好的理解自顶而下的 $dfs$ 的递归过程是如何进行的

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
#define int long long
#define pii pair<int,int>
const int INF = 0x3f3f3f3f;
const int IINF = 0x3f3f3f3f3f3f3f3fLL;
const int mod = 1000000007;
const int maxn = 400007;
const double eps = 1e-6;
inline int read() {
	int x = 0, f = 1;
	char c = getchar();
	while (c < '0' || c > '9') {
		if (c == '-')f = -1;
		c = getchar();
	}
	while (c >= '0' && c <= '9') {
		x = x * 10 + c - '0';
		c = getchar();
	}
	return x * f;
}

int n, f[maxn][2], sz[maxn];
vector<int>G[maxn];
int rt = 1;
int ans[maxn], cnt[maxn];
void dfs(int u, int fa)
{
	bool ok = 1;
	sz[u] = 1;
	for (auto v : G[u])
	{
		if (v == fa)continue;
		dfs(v, u);
		sz[u] += sz[v];
		if (sz[v] > n / 2)ok = 0;
	}
	if (n - sz[u] > n / 2)ok = 0;
	if (ok)rt = u;
}
void dfss(int u, int fa)
{
	sz[u] = 1;
	for (auto v : G[u])
	{
		if (v == fa)continue;
		dfss(v, u);
		sz[u] += sz[v];
		if (sz[v] > n / 2)continue;
		if (sz[v] > f[u][1])
		{
			f[u][1] = sz[v];
			if (f[u][1] > f[u][0])swap(f[u][0], f[u][1]);
		}
	}
}
void GetAns(int u, int fa, int maxx)
{
	for (auto v : G[u])
	{
		if (v == fa)continue;
		//printf("%lld %lld:%lld\n", u,v, maxx);
		if (n - sz[v] <= n / 2)maxx = max(maxx, n - sz[v]);
		int x;
		if (sz[v] != f[u][0])x = max(maxx, f[u][0]);
		else x = max(maxx, f[u][1]);
		if (n - sz[v] - x <= n / 2)ans[v] = 1;
		GetAns(v, u, x);
	}
	
}
signed main() {
	n = read();
	for (int i = 1; i < n; i++)
	{
		int u = read(), v = read();
		G[u].push_back(v);
		G[v].push_back(u);
	}
	dfs(1, -1);
    //debug(rt);
	dfss(rt, -1);
	GetAns(rt, -1, -1);
	ans[rt] = 1;
	for (int i = 1; i <= n; i++)
		printf("%lld ", ans[i]);
	puts("\n");
	return 0;
}
