LIS

问题描述：

给定一个长度为 $n$ 的序列 $A$ ，求出一个最长的 $A$ 的子序列，满足该子序列的后一个元素大于前一个元素。（子序列可以不连续）

例题：

HDU1257-最少拦截系统

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 30007;
const double eps = 1e-6;

int n, dp[maxn], a[maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	while (cin >> n)
	{
		mst(dp);
		for (int i = 1; i <= n; i++)
			cin >> a[i];
		a[0] = 0;
		for (int i = 1; i <= n; i++)
		{
			for (int j = 0; j < i; j++)
			{
				if (a[i] > a[j])
					dp[i] = max(dp[i], dp[j] + 1);
			}
		}
		int ans = 0;
		for (int i = 1; i <= n; i++)
			ans = max(ans, dp[i]);
		cout << ans << endl;
	}
	return 0;
}

变式：

POJ1952BUY LOW, BUY LOWER

思路：最长下降子序列（类比LIS即可）+统计方案数量

如何统计方案数量？

$dp[i]$ — 以第 $i$ 个元素结尾的最长下降子序列长度

$cnt[i]$ — 以第 $i$ 个元素结尾的最优方案个数（不可重复）

在更新时有：

1、当 $dp[i]<dp[j]+1$ 时，$cnt[i]=cnt[j]$

2、当 $dp[i]=dp[j]+1$时，$cnt[i] += cnt[j]$

3、当 $a[i]=a[j]$ 时，把 $a[j]$ 赋值为0，因为第 $j$ 个元素为结尾的最优方案构成的集合，一定会包含以第 $i$ 个元素为 结尾的最优方案构成的集合中

AC代码：

#include<iostream>
#include<cstring>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 998244353;
const int maxn = 300007;
const double eps = 1e-6;

int n, dp[maxn], a[maxn], cnt[maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	mstmin(dp); dp[0] = 0;
	a[0] = INF;
	mst0(cnt); cnt[0] = 1;
	int ans1 = 0;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 0; j < i; j++)
		{
			if (a[i] < a[j])
			{
				if (dp[j] + 1 > dp[i])
				{
					dp[i] = dp[j] + 1;
					cnt[i] = cnt[j];
				}
				else if (dp[j] + 1 == dp[i])cnt[i] += cnt[j];
			}
			else if (a[i] == a[j])cnt[j] = 0;
		}
		if (dp[i] > ans1)ans1 = dp[i];
	}
	int ans2 = 0;
	for (int i = 1; i <= n; i++)
		if (dp[i] == ans1)ans2 += cnt[i];
	cout << ans1 << " " << ans2 << endl;
	return 0;
}

LCS

问题描述：给定一个长度为 $n$ 的序列 $A$ 和一个 长度为 $B$ 的序列，求出一个最长的序列，使得该序列既是 $A$ 的子序列，也是 $B$ 的子序列。

裸题：

HDU1159Common Subsequence

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 1007;
const double eps = 1e-6;

int n, m, dp[maxn][maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	string s1, s2;
	while (cin >> s1 >> s2)
	{
		n = s1.size(); m = s2.size();
		mst(dp);
		for (int i = 1; i <= n; i++)
		{
			for (int j = 1; j <= m; j++)
			{
				dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
				if (s1[i - 1] == s2[j - 1])
					dp[i][j] = max(dp[i][j], dp[i - 1][j - 1] + 1);
			}
		}
		cout << dp[n][m] << endl;
	}
	return 0;
}

LCIS

LIS和LCS的结合

状态设计：$dp[i][j]$ — 以 $A$ 序列前 $i$ 个元素和以 $B[j]$ 结尾的LCIS的长度

例题：HDU1423Greatest Common Increasing Subsequence

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 507;
const double eps = 1e-6;

int n, m, a[maxn], b[maxn];
int dp[maxn][maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	int t;
	cin >> t;
	while (t--)
	{
		cin >> n;
		for (int i = 1; i <= n; i++)
			cin >> a[i];
		cin >> m;
		for (int i = 1; i <= m; i++)
			cin >> b[i];
		mst(dp);
		a[0] = b[0] = (-1) * INF;
		for (int i = 1; i <= n; i++)
		{
			int s = 0;
			for (int j = 1; j <= m; j++)
			{
				dp[i][j] = dp[i - 1][j];
				if (a[i] == b[j])
				{
					if (b[j - 1] < b[j])
						s = max(s, dp[i - 1][j - 1] + 1);
				}
				dp[i][j] = max(s, dp[i][j]);
			}
		}
		int ans = 0;
		for (int i = 1; i <= m; i++)
			ans = max(ans, dp[n][i]);
		cout << ans << endl;
		if (t)cout << endl;
	}
	return 0;
}

数字三角形

最好理解的一类模型，贪心即可。

例题：HDU2571命运

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 1007;
const double eps = 1e-6;

int t, n, m;
int maze[maxn][maxn];
int dp[maxn][maxn];
vector<int> vec[1007];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> t;
	for (int i = 1; i <= 1000; i++) //预处理出1~1000的因子
	{
		vec[i].push_back(1);
		for (int j = 2; j * j <= i; j++)
		{
			if (i % j == 0)
			{
				vec[i].push_back(j);
				if ((i / j) != j)vec[i].push_back((i / j));
			}
		}
	}
	while (t--)
	{
		cin >> n >> m;
		for (int i = 1; i <= n; i++)
			for (int j = 1; j <= m; j++)
				cin >> maze[i][j];
		//边界有一点点麻烦
		mst(dp);
		for (int i = 1; i <= n; i++)
			dp[i][0] = (-1) * INF;
		for (int i = 1; i <= m; i++)
			dp[0][i] = (-1) * INF;
		dp[1][0] = dp[0][1] = 0;
		for (int i = 1; i <= n; i++)
		{
			int mx = (-1) * INF;
			for (int j = 1; j <= m; j++)//三条路选择最优的即可
			{
				mx = max(dp[i - 1][j], dp[i][j - 1]);
				for (int k = 0; k < vec[j].size(); k++)
					mx = max(mx, dp[i][vec[j][k]]);
				dp[i][j] = maze[i][j] + mx;
			}
		}
		cout << dp[n][m] << endl;
	}
	return 0;
}

变式:ACWing275传纸条

解题思路：

1、从$(1,1)到(x,y)$ 假设路径的长度的为 $S$ ，则有如下等式成立

这一等式也是优化状态设计的关键所在（稍做推导即可得出）

2、状态方程设计：$dp[i][j][k]$ — 路径长度为 $i$ ，两条路径的横坐标分别为 $j$ 和 $k$ 的热心值的最大值

3、根据状态方程很好写出状态的转移，有四种方式移动

4、特殊情况：路径进入同一个格子的情况，只需要把保证热心值只加一次即可。因为路径在原来的基础上可以做出修改，使路径上又多了一个人，求出来的值必定会 >= 原来结果。

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 998244353;
const int maxn = 57;
const double eps = 1e-6;

int n, m, a[maxn][maxn];
int dp[maxn * 2][maxn][maxn];
bool judge(int x, int y)
{
	if (x<1 || x>n || y<1 || y>m)return 0;
	return 1;
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= m; j++)
			cin >> a[i][j];
	int L = m + n - 2;
	mst0(dp);
	for (int l = 0; l < L; l++)
	{
		for (int x1 = 1; x1 <= n; x1++)
		{
			for (int x2 = 1; x2 <= n; x2++)
			{
				int y1 = l + 2 - x1, y2 = l + 2 - x2;
				if (!judge(x1, y1) || !judge(x2, y2))continue;
				int cost = a[x1 + 1][y1] + a[x2 + 1][y2];
				if (x1 == x2)cost /= 2;
				dp[l + 1][x1 + 1][x2 + 1] = max(dp[l + 1][x1 + 1][x2 + 1], dp[l][x1][x2] + cost);
				cost = a[x1 + 1][y1] + a[x2][y2 + 1];
				if (x1 + 1 == x2 && y1 == y2 + 1)cost /= 2;
				dp[l + 1][x1 + 1][x2] = max(dp[l + 1][x1 + 1][x2], dp[l][x1][x2] + cost);
				cost = a[x1][y1 + 1] + a[x2 + 1][y2];
				if (x1 == x2 + 1 && y1 + 1 == y2)cost /= 2;
				dp[l + 1][x1][x2 + 1] = max(dp[l + 1][x1][x2 + 1], dp[l][x1][x2] + cost);
				cost = a[x1][y1 + 1] + a[x2][y2 + 1];
				if (x1 == x2)cost /= 2;
				dp[l + 1][x1][x2] = max(dp[l + 1][x1][x2], dp[l][x1][x2] + cost);
			}
		}
	}
	cout << dp[L][n][n] << endl;
	return 0;
}

构造单调数列

例题1：

CF13CSequence

题目大意：给定一个序列，每次操作可以把某个数+1-1。要求把序列变成非降数列的最少的操作次数。

引理：一定存在一种最优方案，修改后的序列的元素都在原序列中出现。

状态设计：$dp[i][j]$ — 修改到前面 $i$ 个元素，并且第 $i$ 个元素不超过原序列中第 $j$ 小的最小操作数

状态转移：

空间优化：从左至右取dp的值可以减少一维空间的花销

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst(a) memset(a,0,sizeof(a))
#define int long long
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 1000000007;
const int maxn = 5007;
const double eps = 1e-6;

int a[maxn], b[maxn], dp[maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i];
		b[i] = a[i];
	}
	sort(b + 1, b + 1 + n);
	mst(dp);
	dp[0] = 1e18;
	//cout << dp[0] << endl;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			dp[j] = min(dp[j - 1], dp[j] + abs(a[i] - b[j]));
		}
	}
	cout << dp[n] << endl;
	return 0;
}

变式：POJ3666Making the Grade

题目大意：给定长度为 $N$ 的序列 $A$ ，构造一个长度为 $N$ 的序列 $B$ ，$B$ 非严格单调且距 $A$ 的绝对距离最小（每个位置上元素之差的绝对值之和）

构造方法：类似于例题1

AC代码：

#include<cstring>
#include<iostream>
#include<algorithm>
#include<cstdlib>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define int long long
const int INF = 0x3f3f3f3f;
const int maxn = 2007;
const double eps = 1e-6;

int n, dp[maxn], a[maxn], b[maxn], c[maxn];
bool cmp(int a, int b) {
	return a > b;
}
int Abs(int a,int b)
{
	if(a>=b)return (a-b);
	else return (b-a);
}
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i];
		b[i] = a[i];
		c[i] = a[i];
	}
	sort(b + 1, b + n + 1);
	mst0(dp);
	dp[0] = 1e18;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			dp[j] = min(dp[j - 1], dp[j] + Abs(a[i] , b[j]));
		}
	}
	int ans = dp[n];
	sort(c + 1, c + 1 + n, cmp);
	mst0(dp);
	dp[0] = 1e18;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			dp[j] = min(dp[j - 1], dp[j] + Abs(a[i] , c[j]));
		}
	}
	ans = min(ans, dp[n]);
	cout << ans << endl;
	return 0;
}

增加维度表示状态

例题2：

AcWing274移动服务

解题思路：

1、确定状态时，思考状态转移需要哪些维度，状态转移需要用到的维度加入到状态设计中来。

2、在本题中，状态的转移明显需要三个员工的位置信息，所以需要把员工位置包含进状态的设计当中来。

3、状态设计：$dp[i][j][k]$ — 完成前 $i$ 个请求，三个员工分别在 $p[i]、j、k$ 三个位置时的最小花销

AC代码：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 998244353;
const int maxn = 2007;
const double eps = 1e-6;

int L, N, cost[207][207];
int dp[1007][207][207];
int a[1007];
bool judge(int x, int y, int z)
{
	if (x == y)return 1;
	if (x == z)return 1;
	if (y == z)return 1;
	return 0;
}
int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> L >> N;
	for (int i = 1; i <= L; i++)
		for (int j = 1; j <= L; j++)
			cin >> cost[i][j];
	for (int i = 1; i <= N; i++)
		cin >> a[i];
	mstmax(dp);
	a[0] = 1;
	dp[0][2][3] = 0;
	for (int i = 0; i < N; i++)
	{
		for (int j = 1; j <= L; j++)
		{
			for (int k = 1; k <= L; k++)
			{
				int x = a[i], p = a[i + 1];
				if (judge(x, j, k))continue;
				int cnt = dp[i][j][k];
				if (!judge(p, j, k))
					dp[i + 1][j][k] = min(dp[i + 1][j][k], cnt + cost[x][p]);
				if (!judge(x, p, k))
					dp[i + 1][x][k] = min(dp[i + 1][x][k], cnt + cost[j][p]);
				if (!judge(x, j, p))
					dp[i + 1][j][x] = min(dp[i + 1][j][x], cnt + cost[k][p]);
			}
		}
	}
	int ans = INF;
	for (int i = 1; i <= L; i++)
	{
		for (int j = 1; j <= L; j++)
		{
			ans = min(ans, dp[N][i][j]);
		}
	}
	cout << ans << endl;
	return 0;
}

	}

}

整除与余数问题

例题1：CFGYM103765F

解题思路：

1、状态设计：$dp[i][j]$ — 前 $i$ 棵树，摘得的橘子数量模 $m$ 为 $j$ 的方案数

2、状态转移：记 $cnt[k]$ 为第 $i$ 棵橘子树取同余 $k$ 的方案数，则可得到转移方程为：

#include<bits/stdc++.h>
using namespace std;
#define FI freopen("in.txt", "r", stdin)
#define FO freopen("out.txt", "w", stdout)
#define mst0(a) memset(a,0,sizeof(a))
#define mstmax(a) memset(a,0x3f,sizeof(a))
#define mstmin(a) memset(a,0xcf,sizeof(a))
#define debug(a) cout<<#a<<"="<<a<<endl;
typedef pair<int, int> pii;
const int INF = 0x3f3f3f3f;
const int mod = 998244353;
const int maxn = 77;
const double eps = 1e-6;

int n, m, k, a[maxn][maxn];
int dp[maxn][maxn][maxn][maxn];
signed main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> n >> m >> k;
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= m; j++)
			cin >> a[i][j];
	memset(dp, -1, sizeof(dp));
	dp[1][0][0][0] = 0;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			int L = min(j+1 , (m / 2));
			for (int cnt = 0; cnt <= L; cnt++)
			{
				for (int rem = 0; rem < k; rem++)
				{
					dp[i][j + 1][cnt][rem] = max(dp[i][j + 1][cnt][rem], dp[i][j][cnt][rem]);
					if (cnt!=0&&dp[i][j][cnt-1][rem]!=-1)
					{
						int nr = (rem + a[i][j+1]) % k;
						dp[i][j+1][cnt][nr] = max(dp[i][j+1][cnt][nr], dp[i][j][cnt - 1][rem] + a[i][j+1]);
					}
				}
				for (int rem = 0; rem < k; rem++)
					dp[i + 1][0][0][rem] = max(dp[i + 1][0][0][rem], dp[i][j+1][cnt][rem]);
			}
		}
	}
	int ans = 0;
	ans = max(ans, dp[n + 1][0][0][0]);
	cout << ans << endl;
	return 0;
}
